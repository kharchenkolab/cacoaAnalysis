# scITD filters samples severely if some cell types miss them. So, we focus only on
# highly represented cell types.
used_types <- c("Astrocytes", "EN-L2-3", "EN-L4", "EN-L5-6", "IN-PV", "IN-SST", 
                "IN-SV2C", "IN-VIP", "OL", "OPC")
